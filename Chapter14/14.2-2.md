MATRIX-CHAIN-MULTIPLY(A, s, i, j):
q <- inf
optimal_idx <- i
for r <- 1 to j-i: // problem size
    k <- i + r
    q <- s[i, k] + s[k+1, j] + p_(i-1) * p_k * p_j
    if q > s[i, j]: // new min
        s[i, j] <- q
        optimal_idx <- k
multiplied[i, j] <- RECTANGULAR-MATRIX-MULTIPLY(multiplied[i, optimal_idx],[optimal_idx+1, j])
return multiplied[i, j]
