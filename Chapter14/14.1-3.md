```plaintext
BOTTOM-UP-CUT-ROD-MODIFIED(p, n):
let r[0:n] and c[0:n] be new arrays
// c[0:n] stores the cost of making the cost
c[0] <-0
for i <- 1 to n: // rod of length i
    q <- -inf // final price
    c <- 0 // cost of cutting
    for j <- 1 to i:
        if q < p[j] + r[i-j] - (1 + c[i-j]):
            q <- p[j] + r[i-j] - (1 + c[i-j])
            c <- c[i-j] + 1
    r[i] <- q
    c[i] <- c
```
